PROJECT(TRIANGLE)

ENABLE_TESTING()

# Default to Debug mode
IF(NOT CMAKE_BUILD_TYPE)
  SET(CMAKE_BUILD_TYPE Debug)
ENDIF(NOT CMAKE_BUILD_TYPE)

IF(APPLE)
  ADD_DEFINITIONS(-DAPPLE)
ENDIF(APPLE)

SET(LIBRARY_OUTPUT_PATH ${TRIANGLE_SOURCE_DIR}/lib)
SET(EXECUTABLE_OUTPUT_PATH ${TRIANGLE_SOURCE_DIR}/bin)

SET(DataDir ${TRIANGLE_SOURCE_DIR}/data)
SET(TestDataDir ${TRIANGLE_SOURCE_DIR}/testdata)
SET(DataSets spx_z noaa_t4 grbm_t5)

SUBDIRS(src)
